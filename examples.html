Examples¶

Here are three topology optimization examples implemented using the samo_ggp framework.

1. Short Cantilever¶

 1import dolfin as df
 2from dolfin_adjoint import Constant, DirichletBC, RectangleMesh, stop_annotating
 3import numpy as np
 4import gemseo
 5from gemseo import create_scenario
 6from samo_ggp.geometry.factory import GeometryFactory
 7from samo_ggp.physics.factory import PhysicsFactory
 8from samo_ggp.gemseo_wrappers.macro_discipline import GGPMacroDiscipline
 9import matplotlib.pyplot as plt
10import os
11
12def run_short_cantilever():
13    L, H = 50.0, 50.0
14    nelx, nely = 50, 50
15    volfrac = 0.4
16    num_components = 10
17
18    # Mesh and Spaces
19    mesh = RectangleMesh(df.Point(0, 0), df.Point(L, H), nelx, nely)
20    V_u = df.VectorFunctionSpace(mesh, "CG", 1)
21    
22    # Boundary Conditions: Left edge fixed
23    def left_boundary(x, on_boundary): return on_boundary and df.near(x[0], 0.0)
24    bc = [DirichletBC(V_u, Constant((0.0, 0.0)), left_boundary)]
25    
26    # Load: Point load on middle right
27    boundaries = df.MeshFunction("size_t", mesh, mesh.topology().dim() - 1)
28    boundaries.set_all(0)
29    class MiddleRightArea(df.SubDomain):
30        def inside(self, x, on_boundary): return df.near(x[0], L, 2.0) and df.near(x[1], H/2.0, 5.0)
31    MiddleRightArea().mark(boundaries, 1)
32    ds_load = df.Measure("ds", domain=mesh, subdomain_data=boundaries)
33    L_rhs_vec = Constant((0.0, -1.0))
34
35    # Initialization
36    mapper = GeometryFactory.create_mapper("2D_Free", mesh=mesh, num_components=num_components)
37    solver = PhysicsFactory.create_solver("Elasticity_2D", V_u=V_u, bc=bc, ds_load=ds_load, L_rhs_vec=L_rhs_vec)
38    x_init = mapper.get_initial_design(L, H)
39    lb = np.array([0.0, 0.0, 5.0, 2.0, -np.pi] * num_components)
40    ub = np.array([L, H, L, H/2, np.pi] * num_components)
41
42    # Discipline and Optimization
43    disc = GGPMacroDiscipline(mapper, solver, x_init, lb, ub, mesh_area=L*H, name="GGP_Short_Cantilever")
44    design_space = gemseo.algos.design_space.DesignSpace()
45    design_space.add_variable("x_vars", size=len(x_init), lower_bound=lb, upper_bound=ub, value=x_init)
46    
47    scenario = create_scenario(disciplines=[disc], objective_name="compliance", design_space=design_space, formulation_name="DisciplinaryOpt")
48    scenario.add_constraint("volume", "ineq", positive=False, value=volfrac)
49    
50    scenario.execute(algo_name="MMA", max_iter=5, max_optimization_step=0.1)
51
52    # --- Post-Processing ---
53    print("Post-processing optimal design...")
54    os.makedirs("results", exist_ok=True)
55    
56    # 1. Retrieve optimal design variables from GEMSEO
57    opt_x = scenario.optimization_result.x_opt
58    
59    # 2. Assign back to FEniCS constants to update the UFL graph
60    for c, v in zip(disc.controls_objs, opt_x):
61        c.assign(float(v))
62        
63    with stop_annotating():
64        # 3. Recompute the density field and project to a Function Space
65        V_rho = df.FunctionSpace(mesh, "CG", 1)
66        rho_opt_ufl = mapper.map_to_density(disc.controls_objs)
67        rho_opt = df.project(rho_opt_ufl, V_rho)
68        rho_opt.rename("Density", "Optimized GGP Density")
69        
70        # 4. Save to Paraview format (.pvd)
71        df.File("results/short_cantilever_optimized.pvd") << rho_opt
72        
73        # 5. Save PNG using Matplotlib
74        plt.figure(figsize=(6, 5))
75        p = df.plot(rho_opt, cmap="Blues")
76        plt.colorbar(p, label="Density $\\rho$")
77        plt.title("Optimized GGP Topology: Short Cantilever")
78        plt.savefig("results/short_cantilever_optimized.png", dpi=300, bbox_inches="tight")
79        print("Results saved to 'results/' directory.")
80
81if __name__ == "__main__":
82    run_short_cantilever()

2. MBB Beam¶

 1import dolfin as df
 2from dolfin_adjoint import Constant, DirichletBC, RectangleMesh, stop_annotating
 3import numpy as np
 4import gemseo
 5from gemseo import create_scenario
 6from samo_ggp.geometry.factory import GeometryFactory
 7from samo_ggp.physics.factory import PhysicsFactory
 8from samo_ggp.gemseo_wrappers.macro_discipline import GGPMacroDiscipline
 9import matplotlib.pyplot as plt
10import os
11
12def run_mbb_beam():
13    # Only modeling half of the beam due to symmetry
14    L, H = 150.0, 50.0
15    nelx, nely = 150, 50
16    volfrac = 0.5
17    num_components = 20
18
19    # Mesh and Spaces
20    mesh = RectangleMesh(df.Point(0, 0), df.Point(L, H), nelx, nely)
21    V_u = df.VectorFunctionSpace(mesh, "CG", 1)
22    
23    # Boundary Conditions: Left edge symmetry (ux=0), Bottom right point supported (uy=0)
24    def left_symmetry(x, on_boundary): return on_boundary and df.near(x[0], 0.0)
25    def bottom_right_support(x, on_boundary): return on_boundary and df.near(x[0], L, 1.0) and df.near(x[1], 0.0, 1.0)
26    
27    bc = [
28        DirichletBC(V_u.sub(0), Constant(0.0), left_symmetry),
29        DirichletBC(V_u.sub(1), Constant(0.0), bottom_right_support, method="pointwise")
30    ]
31    
32    # Load: Top left corner point load
33    boundaries = df.MeshFunction("size_t", mesh, mesh.topology().dim() - 1)
34    boundaries.set_all(0)
35    class TopLeftArea(df.SubDomain):
36        def inside(self, x, on_boundary): return df.near(x[0], 0.0, 2.0) and df.near(x[1], H, 2.0)
37    TopLeftArea().mark(boundaries, 1)
38    ds_load = df.Measure("ds", domain=mesh, subdomain_data=boundaries)
39    L_rhs_vec = Constant((0.0, -1.0))
40
41    # Initialization
42    mapper = GeometryFactory.create_mapper("2D_Free", mesh=mesh, num_components=num_components)
43    solver = PhysicsFactory.create_solver("Elasticity_2D", V_u=V_u, bc=bc, ds_load=ds_load, L_rhs_vec=L_rhs_vec)
44    x_init = mapper.get_initial_design(L, H)
45    lb = np.array([0.0, 0.0, 5.0, 2.0, -np.pi] * num_components)
46    ub = np.array([L, H, L, H/2, np.pi] * num_components)
47
48    # Discipline and Optimization
49    disc = GGPMacroDiscipline(mapper, solver, x_init, lb, ub, mesh_area=L*H, name="GGP_MBB")
50    design_space = gemseo.algos.design_space.DesignSpace()
51    design_space.add_variable("x_vars", size=len(x_init), lower_bound=lb, upper_bound=ub, value=x_init)
52    
53    scenario = create_scenario(disciplines=[disc], objective_name="compliance", design_space=design_space, formulation_name="DisciplinaryOpt")
54    scenario.add_constraint("volume", "ineq", positive=False, value=volfrac)
55    
56    scenario.execute(algo_name="MMA", max_iter=5, max_optimization_step=0.1)
57
58    # --- Post-Processing ---
59    print("Post-processing optimal design...")
60    os.makedirs("results", exist_ok=True)
61    
62    opt_x = scenario.optimization_result.x_opt
63    for c, v in zip(disc.controls_objs, opt_x):
64        c.assign(float(v))
65        
66    with stop_annotating():
67        V_rho = df.FunctionSpace(mesh, "CG", 1)
68        rho_opt_ufl = mapper.map_to_density(disc.controls_objs)
69        rho_opt = df.project(rho_opt_ufl, V_rho)
70        rho_opt.rename("Density", "Optimized GGP Density")
71        
72        df.File("results/mbb_beam_optimized.pvd") << rho_opt
73        
74        plt.figure(figsize=(10, 4))
75        p = df.plot(rho_opt, cmap="Blues")
76        plt.colorbar(p, label="Density $\\rho$")
77        plt.title("Optimized GGP Topology: MBB Beam (Half Model)")
78        plt.savefig("results/mbb_beam_optimized.png", dpi=300, bbox_inches="tight")
79        print("Results saved to 'results/' directory.")
80
81if __name__ == "__main__":
82    run_mbb_beam()
