    "def run_mbb_beam():\n",
    "    # Only modeling half of the beam due to symmetry\n",
    "    L, H = 150.0, 50.0\n",
    "    nelx, nely = 150, 50\n",
    "    volfrac = 0.5\n",
    "    num_components = 20\n",
    "\n",
    "    # Mesh and Spaces\n",
    "    mesh = RectangleMesh(df.Point(0, 0), df.Point(L, H), nelx, nely)\n",
    "    V_u = df.VectorFunctionSpace(mesh, \"CG\", 1)\n",
    "    \n",
    "    # Boundary Conditions: Left edge symmetry (ux=0), Bottom right point supported (uy=0)\n",
    "    def left_symmetry(x, on_boundary): return on_boundary and df.near(x[0], 0.0)\n",
    "    def bottom_right_support(x, on_boundary): return on_boundary and df.near(x[0], L, 1.0) and df.near(x[1], 0.0, 1.0)\n",
    "    \n",
    "    bc = [\n",
    "        DirichletBC(V_u.sub(0), Constant(0.0), left_symmetry),\n",
    "        DirichletBC(V_u.sub(1), Constant(0.0), bottom_right_support, method=\"pointwise\")\n",
    "    ]\n",
    "    \n",
    "    # Load: Top left corner point load\n",
    "    boundaries = df.MeshFunction(\"size_t\", mesh, mesh.topology().dim() - 1)\n",
    "    boundaries.set_all(0)\n",
    "    class TopLeftArea(df.SubDomain):\n",
    "        def inside(self, x, on_boundary): return df.near(x[0], 0.0, 2.0) and df.near(x[1], H, 2.0)\n",
    "    TopLeftArea().mark(boundaries, 1)\n",
    "    ds_load = df.Measure(\"ds\", domain=mesh, subdomain_data=boundaries)\n",
    "    L_rhs_vec = Constant((0.0, -1.0))\n",
    "\n",
    "    # Initialization\n",
    "    mapper = GeometryFactory.create_mapper(\"2D_Free\", mesh=mesh, num_components=num_components)\n",
    "    solver = PhysicsFactory.create_solver(\"Elasticity_2D\", V_u=V_u, bc=bc, ds_load=ds_load, L_rhs_vec=L_rhs_vec)\n",
    "    x_init = mapper.get_initial_design(L, H)\n",
    "    lb = np.array([0.0, 0.0, 5.0, 2.0, -np.pi] * num_components)\n",
    "    ub = np.array([L, H, L, H/2, np.pi] * num_components)\n",
    "\n",
    "    # Discipline and Optimization\n",
    "    disc = GGPMacroDiscipline(mapper, solver, x_init, lb, ub, mesh_area=L*H, name=\"GGP_MBB\")\n",
    "    design_space = gemseo.algos.design_space.DesignSpace()\n",
    "    design_space.add_variable(\"x_vars\", size=len(x_init), lower_bound=lb, upper_bound=ub, value=x_init)\n",
    "    \n",
    "    scenario = create_scenario(disciplines=[disc], objective_name=\"compliance\", design_space=design_space, formulation_name=\"DisciplinaryOpt\")\n",
    "    scenario.add_constraint(\"volume\", \"ineq\", positive=False, value=volfrac)\n",
    "    \n",
    "    scenario.execute(algo_name=\"MMA\", max_iter=50, max_optimization_step=0.1)\n",
    "\n",
    "    # --- Post-Processing ---\n",
    "    print(\"Post-processing optimal design...\")\n",
    "    os.makedirs(\"results\", exist_ok=True)\n",
    "    \n",
    "    opt_x = scenario.optimization_result.x_opt\n",
    "    for c, v in zip(disc.controls_objs, opt_x):\n",
    "        c.assign(float(v))\n",
    "        \n",
    "    with stop_annotating():\n",
    "        V_rho = df.FunctionSpace(mesh, \"CG\", 1)\n",
    "        rho_opt_ufl = mapper.map_to_density(disc.controls_objs)\n",
    "        rho_opt = df.project(rho_opt_ufl, V_rho)\n",
    "        rho_opt.rename(\"Density\", \"Optimized GGP Density\")\n",
    "        \n",
    "        df.File(\"results/mbb_beam_optimized.pvd\") << rho_opt\n",
    "        \n",
    "        plt.figure(figsize=(10, 4))\n",
    "        p = df.plot(rho_opt, cmap=\"Blues\")\n",
    "        plt.colorbar(p, label=\"Density $\\\\rho$\")\n",
    "        plt.title(\"Optimized GGP Topology: MBB Beam (Half Model)\")\n",
    "        plt.savefig(\"results/mbb_beam_optimized.png\", dpi=300, bbox_inches=\"tight\")\n",
    "        print(\"Results saved to 'results/' directory.\")"
   ]
